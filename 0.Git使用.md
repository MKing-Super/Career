windows-cmd

```
ssh-keygen -t rsa	
```

git-bash

```bash
# 初始化（第一次用）
git init

#  把所有有变化的文件都放进暂存区
git add .

# 把暂存区中有变更的文件提交到本地仓库
git commit -m 'my commit'

# 和远程仓库连接
git remote add origin git@github.com:MKing-Super/Career.git

# 把本地仓库和远程仓库关联之后，就可以用git push 命令来推送代码
git push git@github.com:MKing-Super/Career.git
```
