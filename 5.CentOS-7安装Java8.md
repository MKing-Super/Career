# CentOS-7安装Java8

jdk-8u301-linux-x64.tar.gz

```bash
# 创建安装目录
[root@localhost home]# mkdir -p /usr/local/java
# 解压到指定目录
[root@localhost home]# tar -zxvf jdk-8u301-linux-x64.tar.gz -C /usr/local/java/
# 检查解压后的目录
[root@localhost home]# ls /usr/local/java/
# 编辑全局环境变量文件
[root@localhost home]# vi /etc/profile.d/java.sh

export JAVA_HOME=/usr/local/java/jdk1.8.0_301
export JRE_HOME=$JAVA_HOME/jre
export CLASSPATH=.:$JAVA_HOME/lib:$JRE_HOME/lib
export PATH=$PATH:$JAVA_HOME/bin


# 使配置生效
[root@localhost home]# source /etc/profile.d/java.sh

# 验证安装
[root@localhost home]# java -version
[root@localhost home]# javac -version

# 设置默认 Java 版本（可选）
[root@localhost home]# alternatives --install /usr/bin/java java /usr/local/java/jdk1.8.0_301/bin/java 1
[root@localhost home]# alternatives --install /usr/bin/javac javac /usr/local/java/jdk1.8.0_301/bin/javac 1
[root@localhost home]# alternatives --config java   # 选择 JDK 8
[root@localhost home]# alternatives --config javac  # 选择 JDK 8 编译器

# 卸载旧版本（可选）
[root@localhost home]# yum remove java-1.7.0-openjdk  # 移除 OpenJDK 7（如果存在）
```
